158int8_t convertToBcdArray(uint32_t data, uint8_t digits, uint8_t *bcd_number)
159{
160 if (bcd_number == NULL) return -1;
161
162 // Lleno desde el final para mantener orden correcto
163 for (int i = digits - 1; i >= 0; i--) {
164 bcd_number[i] = data % 10; // obtengo último dígito
165 data = data / 10; // elimino ese dígito
166 }
167
168 return 0;
169}

222void bcdToGpio(uint8_t bcd, gpioConf_t *vec)
223{
224 for (int i = 0; i < 4; i++) {
225
226 // Extraigo el bit i del número
227 uint8_t bit = (bcd >> i) & 0x01;
228
229 // Escribo ese bit en el GPIO correspondiente
230 if (bit) {
231 GPIOState(vec[i].pin, 1);
232 } else {
233 GPIOState(vec[i].pin, 0);
234 }
235 }
236}

302void displayNumber(uint32_t data, uint8_t digits,
303 gpioConf_t *bcd_pins,
304 gpioConf_t *digit_pins)
305{
306 uint8_t bcd_array[digits];
307
308 // Convierto el número a array de dígitos
309 convertToBcdArray(data, digits, bcd_array);
310
311 // Recorro cada dígito del display
312 for (int i = 0; i < digits; i++) {
313
314 // Apago todos los dígitos antes de cambiar datos
315 for (int j = 0; j < digits; j++) {
316 GPIOState(digit_pins[j].pin, 0);
317 }
318
319 // Envío el valor BCD del dígito actual
320 bcdToGpio(bcd_array[i], bcd_pins);
321
322 // Activo SOLO el dígito correspondiente
323 GPIOState(digit_pins[i].pin, 1);
324
325 // Pequeño delay para que parezca que no se borran
326 // Permite que el ojo humano perciba el dígito (persistencia visual)
327 vTaskDelay(pdMS_TO_TICKS(DISPLAY_MULTIPLEX_DELAY_MS));
328 }
329}

417 // Mostrar número continuamente mediante multiplexación
